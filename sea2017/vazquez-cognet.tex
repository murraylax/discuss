  \ft{Econometric Model}
  \uncover<+->{
  \begin{block}{Well Done and Thought Out} 
    \bi
    \item Randomized treatment for both online and face-to-face
    \item Accounts for differences in gender, ACT score, year-in-school, interacts these with proctoring
    \item Lack of differences in online section: magnitudes are small and precisely estimated
    \ei
  \end{block}}

  \uncover<+->{
  \begin{block}{Suggestions}
    \bi
    \item Is test anxiety really independent of F2F vs online (accelerated term) decision?
    \item ``Optimal cheating'' w/r to student: Does previous exam performance influence cheating decision?
      \bi
      \item You do see more cheating in later exams. Interact this with previous performance.
      \ei
    \ei
  \end{block} }
}

\frame{
  \ft{Impact of Cheating?}
  \uncover<+->{
  \begin{block}{Important for Understanding ``Optimal Cheating''}
  \bi
  \item Does cheating improve long-run retention?
  \item Are the students cheating themselves?
    \ei
  \end{block}}

  \uncover<+->{
    \begin{block}{Suggested Strategy}
      \bi
      \item Treatment: All midterm exams are proctored
      \item Control: None of midterm exams are proctored
      \item Final exam: All exams are proctored for treatment \& control
      \item Estimate immediate effects of treatment like you do
      \item Estimate effects on final exam
        \ei
    \end{block}
  }
